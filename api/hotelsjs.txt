import fetch from "node-fetch";
import crypto from "crypto";

export default async function handler(req, res) {
  try {
    const { destination, checkIn, checkOut, adults } = req.query;
    const sig = getHotelbedsSignature();

    const url = `https://api.test.hotelbeds.com/hotel-api/1.0/hotels?destinationCode=${destination}&checkIn=${checkIn}&checkOut=${checkOut}&adults=${adults}`;

    const response = await fetch(url, {
      headers: {
        "Api-key": process.env.HOTELBEDS_API_KEY,
        "X-Signature": sig,
        "Accept": "application/json",
      }
    });

    const data = await response.json();
    res.status(200).json(data);
  } catch (err) {
    console.error(err);
    res.status(500).json({ error: "Hotelbeds API 호출 실패" });
  }
}

function getHotelbedsSignature() {
  const ts = Math.floor(Date.now() / 1000);
  return crypto.createHash("sha256")
    .update(process.env.HOTELBEDS_API_KEY + process.env.HOTELBEDS_SECRET + ts)
    .digest("hex");
}
